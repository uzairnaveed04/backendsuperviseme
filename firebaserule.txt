rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ‚úÖ Students - can only access their own document
    match /students/{studentId} {
      allow read, write, update, delete: if request.auth != null && request.auth.uid == studentId;
    }

    // ‚úÖ Supervisors - can only access their own document
    // - Any authenticated user can READ supervisor docs (for listing/selection)
    // - Only the supervisor themself can CREATE/UPDATE/DELETE their own doc (by email)
    match /supervisors/{supervisorId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null &&
                   request.resource.data.email.lower() == request.auth.token.email.lower();
      allow update, delete: if request.auth != null &&
                            resource.data.email.lower() == request.auth.token.email.lower();
    }

    // ‚úÖ PROJECTS COLLECTION (UPDATED)
    match /projects/{projectId} {

      // üîπ READ ACCESS
      allow read: if request.auth != null && (
        resource.data.studentEmail.lower() == request.auth.token.email.lower() ||
        resource.data.supervisorEmail.lower() == request.auth.token.email.lower() ||
        resource.data.status == 'available'
      );

      // üîπ CREATE ACCESS
      allow create: if request.auth != null && (
        request.resource.data.studentEmail.lower() == request.auth.token.email.lower() ||
        request.resource.data.supervisorEmail.lower() == request.auth.token.email.lower()
      );

        // üîπ UPDATE ACCESS (secure)
        allow update: if request.auth != null && (
          resource.data.studentEmail.lower() == request.auth.token.email.lower() ||
          resource.data.supervisorEmail.lower() == request.auth.token.email.lower()
        );
        // üîπ DELETE ACCESS (testing only)
        allow delete: if request.auth != null;
    }

    // ‚úÖ Project Submissions
    match /projectSubmissions/{submissionId} {
      allow read, write, update, delete: if request.auth != null && (
        resource.data.studentUID == request.auth.uid || 
        resource.data.supervisorEmail.lower() == request.auth.token.email.lower()
      );
    }

    // ‚úÖ Project Requests
    match /projectRequests/{requestId} {
      allow read, update, delete: if request.auth != null && (
        resource.data.studentUID == request.auth.uid || 
        resource.data.supervisorEmail.lower() == request.auth.token.email.lower() ||
        resource.data.studentEmail.lower() == request.auth.token.email.lower()
      );
      allow create: if request.auth != null && (
        request.resource.data.studentEmail.lower() == request.auth.token.email.lower() ||
        request.resource.data.supervisorEmail.lower() == request.auth.token.email.lower()
      );
    }

    // ‚úÖ General Requests
    match /requests/{requestId} {
      allow read, write, update, delete: if request.auth != null && (
        resource.data.studentEmail.lower() == request.auth.token.email.lower() || 
        resource.data.supervisorEmail.lower() == request.auth.token.email.lower()
      );
    }

    // ‚úÖ Messages
    match /messages/{messageId} {
      allow read: if request.auth != null && (
        resource.data.senderEmail.lower() == request.auth.token.email.lower() ||
        resource.data.receiverEmail.lower() == request.auth.token.email.lower()
      );
      allow create: if request.auth != null && (
        request.resource.data.senderEmail.lower() == request.auth.token.email.lower()
      );
      allow update, delete: if request.auth != null && (
        resource.data.senderEmail.lower() == request.auth.token.email.lower() ||
        resource.data.receiverEmail.lower() == request.auth.token.email.lower()
      );
    }

    // ‚úÖ Chats
    match /chats/{chatId} {
      allow read: if request.auth != null && (
        chatId.matches('(?i).*' + request.auth.token.email + '.*')
      );
      
      allow create: if request.auth != null && (
        chatId.matches('(?i).*' + request.auth.token.email + '.*')
      ) && (
        request.resource.data.senderId.matches('(?i)^' + request.auth.token.email + '$')
      );
      
      allow update, delete: if request.auth != null && (
        chatId.matches('(?i).*' + request.auth.token.email + '.*')
      );

      match /messages/{messageId} {
        allow read, write, update, delete: if request.auth != null && (
          chatId.matches('(?i).*' + request.auth.token.email + '.*')
        );
      }
    }

    // ‚úÖ Tasks
    match /tasks/{taskId} {
      allow read: if request.auth != null && (
        resource.data.assignedTo == request.auth.token.email.lower() ||
        resource.data.createdBy == request.auth.token.email.lower()
      );

      allow update: if request.auth != null && (
        (resource.data.assignedTo == request.auth.token.email.lower()) ||
        (resource.data.createdBy == request.auth.token.email.lower())
      );

      allow create: if request.auth != null &&
                   request.resource.data.createdBy == request.auth.token.email.lower();

      allow delete: if request.auth != null &&
                   resource.data.createdBy == request.auth.token.email.lower();
    }

    // ‚úÖ Reminders
    match /reminders/{reminderId} {
      allow read: if request.auth != null && (
        resource.data.studentEmail == request.auth.token.email.lower() ||
        resource.data.supervisorEmail == request.auth.token.email.lower()
      );

      allow update: if request.auth != null && (
        (resource.data.studentEmail == request.auth.token.email.lower()) ||
        (resource.data.supervisorEmail == request.auth.token.email.lower())
      );

      allow create: if request.auth != null &&
                    request.resource.data.supervisorEmail == request.auth.token.email.lower();

      allow delete: if request.auth != null && (
        resource.data.studentEmail == request.auth.token.email.lower() ||
        resource.data.supervisorEmail == request.auth.token.email.lower()
      );
    }

    // ‚úÖ Reminder History
    match /reminderHistory/{historyId} {
      allow read: if request.auth != null && (
        resource.data.studentEmail == request.auth.token.email.lower() ||
        resource.data.supervisorEmail == request.auth.token.email.lower()
      );

      allow create: if request.auth != null && (
        request.resource.data.studentEmail == request.auth.token.email.lower() ||
        request.resource.data.supervisorEmail == request.auth.token.email.lower()
      );

      allow update, delete: if false;
    }

    // ‚úÖ Weekly Logs
    match /weeklyLogs/{logId} {
      allow read: if request.auth != null && (
        resource.data.studentEmail == request.auth.token.email ||
        resource.data.supervisorEmail == request.auth.token.email
      );
      allow create: if request.auth != null;
    }

    // ‚úÖ Required Subjects
    match /requiredSubjectsV2/{docId} {
      allow read, write: if request.auth != null;
      match /studentSubjects/{studentId} {
        allow read, write: if request.auth != null && request.auth.uid == studentId;
      }
    }

    // ‚úÖ Feedback Evaluation
    match /feedbackEvaluation/{docId} {
      allow read, write: if request.auth != null;
    }

    // ‚úÖ Supervisors Chat
    match /supervisorschat/{docId} {
      allow read, write, update, delete: if request.auth != null && (
        resource.data.supervisorEmail == request.auth.token.email ||
        resource.data.participants[request.auth.token.email] == true
      );
    }

    // ‚ùå Deny everything else
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
